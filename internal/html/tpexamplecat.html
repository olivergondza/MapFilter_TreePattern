cat utility filter

Following example shows how the pattern for cat (http://linux.die.net/man/1/cat) utility may look like.

<?xml version="1.0" encoding="UTF-8" standalone="no"?>
<!DOCTYPE patterns SYSTEM "TreePattern053.dtd">
<!-- [TreePattern_Cat] -->
<patterns>
  <pattern>
    <one>

      <key flag="show_help" name="help" />

      <key flag="show_version" name="version" />

      <some flag="perform_action">

        <!-- ignote option and set assert if present -->
        <key name="u">
          <value pattern="/(?!)/" assert="deprecated" />
        </key>

        <!-- translate long options to its short versions -->
        <alias name="number-nonblank">
          <key name="b" />
        </alias>

        <alias name="show-ends">
          <key name="E" />
        </alias>

        <alias name="number">
          <key name="n" />
        </alias>

        <alias name="squeeze-blank">
          <key name="s" />
        </alias>

        <alias name="show-tabs">
          <key name="T" />
        </alias>

        <alias name="show-nonprinting">
          <key name="v" />
        </alias>

        <!-- expand aggregative options -->
        <alias name="show-all">
          <key name="v" />
          <key name="E" />
          <key name="T" />
        </alias>

        <alias name="A">
          <key name="v" />
          <key name="E" />
          <key name="T" />
        </alias>

        <alias name="e">
          <key name="v" />
          <key name="E" />
        </alias>

        <alias name="t">
          <key name="v" />
          <key name="T" />
        </alias>

        <!-- allowed short options -->
        <some flag="number">
          <key name="b" flag="number_nonblank" />
          <key name="n" flag="number_all" />
        </some>

        <key name="E" flag="show_ends" />
        <key name="s" flag="squeeze_blank" />
        <key name="T" flag="show_tabs" />
        <key name="v" flag="show_nonprinting" />
      </some>

    </one>
  </pattern>
</patterns>
<!-- [TreePattern_Cat] -->

There is a list of input arrays on the first row of each set and the result arrays on the second. Flags can be found on the third row.
